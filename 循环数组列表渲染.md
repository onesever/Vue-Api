```JS
// 完整demo
<template>
  <div class="test">
    <div>测试</div>
    <button @click="test">测试this.$message.error</button>
    <div>v-for</div>
    <ul>
      <li v-for="item in arr">
        <p v-for="items in item.img1">{{items.imgurl}}</p>  // 循环数组成功

</template>

        ]}, {
          'img2':[
            {'imgurl': '1'},
            {'order': '2',},
            {'info': '3'},
            {'radio':'4'}
        ]}
        ]
      }
    },
    methods:{
      test(){
        console.log(this.$message)
      }
    },
    mounted () {

    }
  }
</script>
```
